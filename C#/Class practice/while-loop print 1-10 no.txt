//wap to print no from 1 to 10 using while loop

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace while_loop_print_1_10_no
{
     class Program
    {
        static void Main(string[] args)
        {
            
            int n = 1;
            while(n<=10)
            {
                Console.WriteLine(n);
                n++;
            }
        }
    }
}
